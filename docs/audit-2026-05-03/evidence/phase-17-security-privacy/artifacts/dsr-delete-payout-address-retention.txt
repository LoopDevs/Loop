     8	 * **What "delete" means here.** ADR-009 makes the credit ledger
     9	 * append-only; tax / regulatory rules (UK FCA + IRS) require we
    10	 * retain the underlying transaction record. So "delete" is
    11	 * **anonymisation** rather than a hard row removal:
    12	 *
    13	 *   - `users.email` is replaced with a synthetic
    14	 *     `deleted-{userId}@deleted.loopfinance.io` placeholder so the
    15	 *     row remains FK-consistent with the ledger but no longer carries
    16	 *     a real PII anchor.
    17	 *   - `users.stellar_address` is set NULL.
    18	 *   - `users.ctx_user_id` is set NULL.
    19	 *   - `user_identities` rows are deleted (Google/Apple OAuth links).
    20	 *   - All refresh tokens are revoked so any session in flight is
    21	 *     immediately invalidated.
    22	 *
    23	 * Ledger rows (`credit_transactions` / `orders` / `pending_payouts`)
    24	 * are RETAINED because the FK is `onDelete: 'restrict'` — we'd
    25	 * violate the append-only invariant by deleting them, and we need
    26	 * them for tax reporting per A2-1923. They no longer link to a real
    27	 * person after the email scrub.
    28	 *
    29	 * **Pre-conditions.** Refuse to anonymise when:
    30	 *   - any `pending_payouts` row is in `pending` or `submitted` state
    31	 *     — money is in flight; user must wait or contact support
    32	 *   - any `orders` row is in `pending_payment` / `paid` / `procuring`
    86	export async function deleteUserViaAnonymisation(userId: string): Promise<DsrDeleteResult> {
    87	  // Block: any pending or submitted payout means money in flight.
    88	  // We can't guarantee the right end-state for that money post-
    89	  // anonymisation (e.g. an `op_no_trust` failure that needs the
    90	  // user's Stellar address — which we just nulled).
    91	  const blockingPayouts = await db
    92	    .select({ id: pendingPayouts.id })
    93	    .from(pendingPayouts)
    94	    .where(
    95	      and(
    96	        eq(pendingPayouts.userId, userId),
    97	        inArray(pendingPayouts.state, [
    98	          'pending',
    99	          'submitted',
   100	        ] satisfies (typeof PAYOUT_STATES)[number][]),
   101	      ),
   102	    )
   103	    .limit(1);
   104	  if (blockingPayouts.length > 0) {
   105	    return { ok: false, blockedBy: 'pending_payouts' };
   106	  }
   107	
   108	  // Block: orders mid-fulfilment. ADR 010 transitions are
   109	  // pending_payment → paid → procuring → fulfilled / failed.
   110	  // Anonymising during procurement could drop the email the CTX
   111	  // proxy flow relies on for order-attribution.
   112	  const blockingOrders = await db
   113	    .select({ id: orders.id })
   114	    .from(orders)
   115	    .where(
   116	      and(
   117	        eq(orders.userId, userId),
   118	        inArray(orders.state, [
   119	          'pending_payment',
   120	          'paid',
   121	          'procuring',
   122	        ] satisfies (typeof ORDER_STATES)[number][]),
   123	      ),
   124	    )
   125	    .limit(1);
   126	  if (blockingOrders.length > 0) {
   127	    return { ok: false, blockedBy: 'in_flight_orders' };
   128	  }
   129	
   130	  await db.transaction(async (tx) => {
   131	    // OAuth identity links — deleting these means a re-auth via
   132	    // Google/Apple under the same provider_sub spawns a fresh user
   133	    // row instead of resolving to the anonymised one.
   134	    await tx.delete(userIdentities).where(eq(userIdentities.userId, userId));
   135	
   136	    // Email gets the synthetic placeholder. ctx_user_id and
   137	    // stellar_address null out — both are PII anchors.
   138	    await tx
   139	      .update(users)
   140	      .set({
   141	        email: deletedEmailFor(userId),
   142	        ctxUserId: null,
   143	        stellarAddress: null,
   144	      })
   145	      .where(eq(users.id, userId));
   146	  });
