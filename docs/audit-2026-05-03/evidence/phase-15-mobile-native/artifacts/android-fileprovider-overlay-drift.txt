    29	  try {
    30	    const { Filesystem, Directory } = await import('@capacitor/filesystem');
    31	    const parsed = parseBase64DataUrl(imageUrl);
    32	    if (parsed === null) return null;
    33	    // A2-1213: prefix the filename with `share/` so it lands in
    34	    // `<cache>/share/`, the only directory the Android FileProvider
    35	    // grants access to (see `apps/mobile/native-overlays/android/app/
    36	    // src/main/res/xml/file_paths.xml`). Capacitor's writeFile creates
    37	    // intermediate directories automatically.
    38	    const scopedPath = `share/${filename}`;
    39	    // No `encoding` field — the plugin interprets the data as
    40	    // base64 when the option is omitted, which is exactly what we
    41	    // want for binary PNG payloads. Passing `Encoding.UTF8` would
    42	    // write the base64 string as literal text instead of decoding.
    43	    await Filesystem.writeFile({
    44	      path: scopedPath,
    45	      data: parsed.base64,
    46	      directory: Directory.Cache,
    47	      recursive: true,
    48	    });
    49	    const { uri } = await Filesystem.getUri({
    50	      path: scopedPath,
    51	      directory: Directory.Cache,
    52	    });
    53	    return uri;
    54	  } catch {
    55	    return null;
    56	  }
    57	}
